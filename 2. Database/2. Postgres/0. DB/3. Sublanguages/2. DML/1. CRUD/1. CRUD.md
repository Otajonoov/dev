1. Create
```sql
-- Bitta qator
INSERT INTO aircrafts (aircraft_code, model, range)
VALUES ('SU9', 'Sukhoi SuperJet-100', 3000);

-- Bir nechta qator
INSERT INTO aircrafts VALUES 
    ('773', 'Boeing 777-300', 11100),
    ('320', 'Airbus A320-200', 5700);
```


2. Read
```sql
-- Barcha ustunlar
SELECT * FROM aircrafts;

-- Maxsus ustunlar va tartib
SELECT model, range 
FROM aircrafts 
ORDER BY range DESC;

-- Shart bilan
SELECT model, range 
FROM aircrafts 
WHERE range >= 4000 AND range <= 6000;
```

3. Update
```sql
UPDATE aircrafts 
SET range = 3500 
WHERE aircraft_code = 'SU9';

-- Hisoblash bilan
UPDATE aircrafts 
SET range = range * 2 
WHERE aircraft_code = 'SU9';
```

4. Delete
```sql
-- Shart bilan
DELETE FROM aircrafts 
WHERE aircraft_code = 'CN1';

-- Murakkab shart
DELETE FROM aircrafts 
WHERE range > 10000 OR range < 3000;
```
