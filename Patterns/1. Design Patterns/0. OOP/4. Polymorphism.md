# Polimorfizm (Polymorphism)

## Umumiy tushuntirish

**Polimorfizm** — dasturning **bir xil nomdagi operatsiyalarni** chaqirishda turli xil amalga oshirishlarni tanlash qobiliyati.

"Polymorphism" = poly (ko'p) + morphe (shakl) → **Ko'p shakllilik**

**Mohiyati:** Bir interfeys — ko'p xatti-harakat.

```
          ┌─────────────────┐
          │   Shakl (ABC)   │
          │ + maydon()      │   ← Bir xil nom
          └────────┬────────┘
                   │
        ┌──────────┼──────────┐
        │          │          │
   ┌────┴──┐  ┌───┴───┐  ┌───┴────┐
   │Doira  │  │To'g'r │  │Uchbur. │
   │ πr²   │  │  a*b  │  │ ½*a*h  │   ← Turli xil natija
   └───────┘  └───────┘  └────────┘
```

**Polimorfizmning ikki turi:**

| Tur | Qachon | Misol |
|-----|--------|-------|
| **Compile-time** | Kompilyatsiya vaqtida | Method overloading (Go da yo'q) |
| **Runtime** | Dastur ishlayotgan vaqtda | Method overriding, Interface |

## Python

Python da polimorfizm ikki usulda:
1. **Duck typing** — "Agar g'oz kabi yurib, g'oz kabi gapirsa — g'oz!" (interfeys tekshiruvuz)
2. **ABC** — rasmiy interfeys orqali

```python
from abc import ABC, abstractmethod
import math

# 1. ABC orqali polimorfizm
class Shakl(ABC):
    @abstractmethod
    def maydon(self) -> float:
        pass

    @abstractmethod
    def perimetr(self) -> float:
        pass

    def tavsif(self) -> str:
        return f"{self.__class__.__name__}: maydon={self.maydon():.2f}, perimetr={self.perimetr():.2f}"


class Doira(Shakl):
    def __init__(self, radius: float):
        self.radius = radius

    def maydon(self) -> float:
        return math.pi * self.radius ** 2

    def perimetr(self) -> float:
        return 2 * math.pi * self.radius


class Tortburchak(Shakl):
    def __init__(self, eni: float, boyi: float):
        self.eni = eni
        self.boyi = boyi

    def maydon(self) -> float:
        return self.eni * self.boyi

    def perimetr(self) -> float:
        return 2 * (self.eni + self.boyi)


class Uchburchak(Shakl):
    def __init__(self, a: float, b: float, c: float):
        self.a, self.b, self.c = a, b, c

    def maydon(self) -> float:
        s = self.perimetr() / 2
        return math.sqrt(s * (s - self.a) * (s - self.b) * (s - self.c))

    def perimetr(self) -> float:
        return self.a + self.b + self.c


# Polimorfizm — bir tsikl, turli xatti-harakat
shakllar: list[Shakl] = [
    Doira(5),
    Tortburchak(4, 6),
    Uchburchak(3, 4, 5),
]

for shakl in shakllar:
    print(shakl.tavsif())
```

**Natija:**
```
Doira: maydon=78.54, perimetr=31.42
Tortburchak: maydon=24.00, perimetr=20.00
Uchburchak: maydon=6.00, perimetr=12.00
```

```python
# 2. Duck typing — Python ga xos polimorfizm
# Interfeys talab qilinmaydi — faqat metod nomi mos kelsa yetarli

class It:
    def salom(self) -> str:
        return "Vov!"

class Mushuk:
    def salom(self) -> str:
        return "Miyov!"

class Robot:
    def salom(self) -> str:
        return "Beep boop!"

class Inson:
    def salom(self) -> str:
        return "Salom!"

# Barcha turlar bilan ishlaydi — ABC yo'q, meros yo'q
def salomlash(narsa) -> None:
    print(narsa.salom())  # duck typing

for narsa in [It(), Mushuk(), Robot(), Inson()]:
    salomlash(narsa)
```

```python
# 3. Operator overloading — Python da qo'shimcha polimorfizm
class Vektor:
    def __init__(self, x: float, y: float):
        self.x = x
        self.y = y

    def __add__(self, other: "Vektor") -> "Vektor":
        return Vektor(self.x + other.x, self.y + other.y)

    def __mul__(self, skalyar: float) -> "Vektor":
        return Vektor(self.x * skalyar, self.y * skalyar)

    def __str__(self) -> str:
        return f"Vektor({self.x}, {self.y})"


v1 = Vektor(1, 2)
v2 = Vektor(3, 4)
print(v1 + v2)    # Vektor(4, 6) — __add__ chaqirildi
print(v1 * 3)     # Vektor(3, 6) — __mul__ chaqirildi
```

## Go

Go da polimorfizm faqat **interface** orqali amalga oshiriladi. Operator overloading va method overloading yo'q.

```go
package main

import (
    "fmt"
    "math"
)

// Interface — polimorfizm uchun shartnoma
type Shakl interface {
    Maydon() float64
    Perimetr() float64
}

type Doira struct {
    Radius float64
}

func (d Doira) Maydon() float64    { return math.Pi * d.Radius * d.Radius }
func (d Doira) Perimetr() float64  { return 2 * math.Pi * d.Radius }

type Tortburchak struct {
    Eni, Boyi float64
}

func (t Tortburchak) Maydon() float64   { return t.Eni * t.Boyi }
func (t Tortburchak) Perimetr() float64 { return 2 * (t.Eni + t.Boyi) }

type Uchburchak struct {
    A, B, C float64
}

func (u Uchburchak) Maydon() float64 {
    s := u.Perimetr() / 2
    return math.Sqrt(s * (s - u.A) * (s - u.B) * (s - u.C))
}

func (u Uchburchak) Perimetr() float64 { return u.A + u.B + u.C }

// Umumiy funksiya — Shakl interfeysi orqali polimorfizm
func Tavsif(s Shakl) string {
    return fmt.Sprintf("%T: maydon=%.2f, perimetr=%.2f", s, s.Maydon(), s.Perimetr())
}

func main() {
    shakllar := []Shakl{
        Doira{Radius: 5},
        Tortburchak{Eni: 4, Boyi: 6},
        Uchburchak{A: 3, B: 4, C: 5},
    }

    for _, s := range shakllar {
        fmt.Println(Tavsif(s)) // Bir funksiya — turli natijalar
    }
}
```

**Natija:**
```
main.Doira: maydon=78.54, perimetr=31.42
main.Tortburchak: maydon=24.00, perimetr=20.00
main.Uchburchak: maydon=6.00, perimetr=12.00
```

```go
// Type assertion — aniq turni tekshirish
func ShaklHaqida(s Shakl) {
    // Aniq tur aniqlash
    switch v := s.(type) {
    case Doira:
        fmt.Printf("Doira, radius=%.1f\n", v.Radius)
    case Tortburchak:
        fmt.Printf("To'rtburchak, %g×%g\n", v.Eni, v.Boyi)
    case Uchburchak:
        fmt.Printf("Uchburchak, tomonlar: %g, %g, %g\n", v.A, v.B, v.C)
    default:
        fmt.Printf("Noma'lum shakl: %T\n", v)
    }
}

// Bir interfeys — ko'p xatti-harakat
type Notifier interface {
    Notify(xabar string)
}

type EmailNotifier struct{ Manzil string }
type SMSNotifier struct{ Telefon string }
type PushNotifier struct{ Token string }

func (e EmailNotifier) Notify(x string) { fmt.Printf("Email → %s: %s\n", e.Manzil, x) }
func (s SMSNotifier) Notify(x string)   { fmt.Printf("SMS → %s: %s\n", s.Telefon, x) }
func (p PushNotifier) Notify(x string)  { fmt.Printf("Push → %s: %s\n", p.Token, x) }

func XabarYuborish(notifiers []Notifier, xabar string) {
    for _, n := range notifiers {
        n.Notify(xabar) // Polimorfizm
    }
}

func main() {
    notifiers := []Notifier{
        EmailNotifier{Manzil: "ali@example.com"},
        SMSNotifier{Telefon: "+998901234567"},
        PushNotifier{Token: "abc123"},
    }
    XabarYuborish(notifiers, "Buyurtmangiz tayyor!")
}
```

**Python vs Go — polimorfizm farqi:**

| | Python | Go |
|-|--------|----|
| Usuli | Duck typing + ABC | Faqat Interface |
| Operator overloading | ✅ `__add__`, `__mul__` | ❌ yo'q |
| Method overloading | ❌ yo'q (default args) | ❌ yo'q |
| Implicit/Explicit | Duck typing — implicit | Interface — ham implicit |
| Type switch | `isinstance()` | `switch v := x.(type)` |
